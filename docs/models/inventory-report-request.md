# InventoryReportRequest

## Example Usage

```typescript
import { InventoryReportRequest } from "lapyme/models";

let value: InventoryReportRequest = {
  source: "inventory",
  period: {
    startDate: new Date("2026-01-01"),
    endDate: new Date("2026-03-31"),
  },
  dimensions: [
    "product_metafield:season",
  ],
  measures: [],
};
```
